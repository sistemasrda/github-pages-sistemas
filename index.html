body { font-family: Arial, sans-serif; } form, div { margin-bottom: 20px; } input, textarea { display: block; margin: 10px 0; padding: 8px; width: 300px; } button { padding: 10px 15px; } document.getElementById('inventory-form').addEventListener('submit', function(event) { event.preventDefault(); const formData = new FormData(event.target); const inventoryItem = {}; formData.forEach((value, key) => { inventoryItem[key] = value; }); let inventory = JSON.parse(localStorage.getItem('inventory')) || []; inventory.push(inventoryItem); localStorage.setItem('inventory', JSON.stringify(inventory)); displayDashboard(inventory); event.target.reset(); }); function displayDashboard(inventory) { const dashboard = document.getElementById('dashboard'); dashboard.innerHTML = ''; inventory.forEach(item => { const itemDiv = document.createElement('div'); itemDiv.textContent = `${item['Nombre de usuario']} - ${item['Número de empleado']} - ${item['Tipo de equipo']}`; dashboard.appendChild(itemDiv); }); } document.getElementById('search-btn').addEventListener('click', function() { const searchValue = document.getElementById('search').value.toLowerCase(); const inventory = JSON.parse(localStorage.getItem('inventory')) || []; const filteredInventory = inventory.filter(item => item['Nombre de usuario'].toLowerCase().includes(searchValue) || item['Número de empleado'].toLowerCase().includes(searchValue) ); displayDashboard(filteredInventory); });
